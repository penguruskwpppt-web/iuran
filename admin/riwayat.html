async function cekTunggakanByRumah() { const noRumah = prompt("Masukkan nomor rumah yang ingin dicek tunggakannya:", "") if (!noRumah) return try { // 1. Ambil data warga (pemilik) const { data: warga, error: errWarga } = await supabase .from('warga') .select('no_rumah, nama_pemilik, iuran_per_bulan, kondisi_terakhir') .eq('no_rumah', noRumah) .single() if (errWarga || !warga) { alert(`Nomor rumah ${noRumah} tidak ditemukan.`) return } // 2. Ambil riwayat terbaru yang berlaku sampai hari ini const today = new Date().toISOString().split('T')[0]; const { data: riwayat, error: errRiwayat } = await supabase .from('riwayat_kondisi') .select('*') .eq('no_rumah', noRumah) .lte('berlaku_sejak', today) .order('berlaku_sejak', { ascending: false }) .limit(1); // Gunakan data riwayat jika ada, jika tidak gunakan data warga let namaPenghuni = warga.nama_pemilik; let iuranPerBulan = warga.iuran_per_bulan; let statusRumah = warga.kondisi_terakhir; if (riwayat && riwayat.length > 0) { const r = riwayat[0]; namaPenghuni = r.nama_penghuni || namaPenghuni; iuranPerBulan = r.iuran_per_bulan; statusRumah = r.kondisi; } if (statusRumah === 'Dibebaskan') { alert(`🏠 Rumah ${noRumah} (${namaPenghuni})\n\nStatus: DIBEBASKAN\nAnda tidak dikenakan iuran.`) return } // 3. Ambil semua iuran yang sudah dibayar const { data: payments } = await supabase .from('transaksi_iuran') .select('bulan') .eq('no_rumah', noRumah) const paidMonths = new Set() payments?.forEach(p => { const date = new Date(p.bulan) paidMonths.add(`${date.getFullYear()}-${date.getMonth()+1}`) }) // 4. Tentukan periode perhitungan: 1 November 2025 sampai bulan berjalan const startDate = new Date(2025, 10, 1) const currentDate = new Date() const currentYear = currentDate.getFullYear() const currentMonth = currentDate.getMonth() + 1 const months = [] let year = startDate.getFullYear() let month = startDate.getMonth() + 1 while (year < currentYear || (year === currentYear && month <= currentMonth)) { months.push({ year, month }) month++ if (month > 12) { month = 1; year++ } } // 5. Hitung tunggakan const tunggakan = [] let totalNominal = 0 months.forEach(m => { const key = `${m.year}-${m.month}` if (!paidMonths.has(key)) { tunggakan.push(`${m.month}/${m.year}`) totalNominal += iuranPerBulan } }) // 6. Tampilkan hasil let pesan = `🏠 Rumah ${noRumah}\nPenghuni: ${namaPenghuni}\n` if (statusRumah === 'Kosong') { pesan += `Status: KOSONG (iuran Rp ${formatRupiah(iuranPerBulan)}/bln)\n\n` } else { pesan += `Status: BERPENGHUNI (iuran Rp ${formatRupiah(iuranPerBulan)}/bln)\n\n` } if (tunggakan.length === 0) { pesan += `✅ *Tidak Ada Tunggakan* (Periode Nov 2025 – ${bulanSekarang()})\nTerima kasih telah membayar iuran tepat waktu. 🙏` } else { pesan += `⚠️ *Tunggakan ${tunggakan.length} bulan*:\n${tunggakan.join(', ')}\nTotal: *${formatRupiah(totalNominal)}*\n\n` if (tunggakan.length >= 2) { pesan += `⚠️ *PERINGATAN!* Jika tunggakan >2 bulan, layanan kebersihan akan dihentikan sementara.` } else { pesan += `Segera lunasi tunggakan Anda.` } } alert(pesan.replace(/\*/g, '')) } catch (err) { console.error(err) alert('Gagal memeriksa tunggakan.') } }
